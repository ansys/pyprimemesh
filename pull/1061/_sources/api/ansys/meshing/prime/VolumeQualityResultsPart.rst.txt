   :param part_id: Id of the part for which quality is computed.
   :type part_id: int, optional

   :type quality_limit: float, optional
   :param n_found: Number of failed cells.
   :type n_found: int, optional

   :type max_quality: float, optional
   :param min_quality: Minimum value of quality measure.
   :type min_quality: float, optional
   :param json_data: JSON dictionary to create a ``VolumeQualityResultsPart`` object with provided parameters.
